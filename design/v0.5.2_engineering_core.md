## ✅ Trie构建

```csharp
public class IdiomTrie
{
    private IdiomTrieNode root = new();

    public void Insert(IdiomRecipe recipe)
    {
        var node = root;
        foreach (var letter in recipe.pattern)
        {
            if (!node.children.ContainsKey(letter))
                node.children[letter] = new IdiomTrieNode();

            node = node.children[letter];
        }

        node.isEnd = true;
        node.recipe = recipe;
    }
}
```

---

## ✅ O(k)匹配（核心优化）

```csharp
public IdiomRecipe Match(List<string> input)
{
    var node = root;

    foreach (var letter in input)
    {
        if (!node.children.ContainsKey(letter))
            return null;

        node = node.children[letter];
    }

    return node.isEnd ? node.recipe : null;
}
```

private string DrawFromBag()
{
    if (bag.Count == 0)
        Reshuffle();

    var letter = bag[0];
    bag.RemoveAt(0);
    return letter;
}

# 五、玩家构筑系统（核心状态）

```csharp
public class PlayerState
{
    public List<string> letters = new();
    public List<IdiomRecipe> unlockedIdioms = new();

    public Dictionary<string, float> idiomBuffs = new();

    public void AddLetter(string letter)
    {
        letters.Add(letter);
    }

    public void UnlockIdiom(IdiomRecipe recipe)
    {
        unlockedIdioms.Add(recipe);
    }

    public void BuffIdiom(string id, float mul)
    {
        if (!idiomBuffs.ContainsKey(id))
            idiomBuffs[id] = 1f;

        idiomBuffs[id] *= mul;
    }
}
```
